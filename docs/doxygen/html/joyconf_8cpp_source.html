138 int debouncer(int button){
139 
140  if(test){
141  reading = digitalRead_em();
142  //Serial.print("state: ");
143  //Serial.println(reading);
144  }else{
145  // read the state of the switch into a local variable:
146  reading = digitalRead(button);
147  }
148 
149  /* debouncing function for multiple buttons is hard to implement.
150  * It requires an array where to store the last states to compare to the last
151  * input. By using a multiplexer the input of several buttons is provided
152  * on the same port. This hardware configuration make the software even more difficult.
153  * I should have a timer for every input. The esiest way is a hardware debouncing using
154  * resistors and capacitor or the most advance MC14490P.
155  *
156  * An alternative should be the usage of delay, but this function stop the execution
157  * and introduce lag in the joystick movements.
158  *
159  * For this reason this function is not implemented yet and it return the input read
160  * directly from the port.
161  */
162 
163  return reading;
164 
165 }
